class Solution {
    public int findPeakElement(int[] nums) {
        if(nums.length==1){
            return 0;
        }
        for(int i=1;i<nums.length-1;i++){
            if(nums[i-1]<nums[i] && nums[i]>nums[i+1]){
                return i;
            }
        }
        int res = 0;
        if(nums[0]>nums[nums.length-1]){
            res = 0;
        }
        else{
            res = nums.length-1;
        }
        return res ;
    }
}
